-- 1. Create the Orders Table
CREATE TABLE Orders (
    OrderID INT,
    OrderAmount VARCHAR(20), -- Stored as string to allow 'dirty' data
    OrderDate DATETIME2(6)
);

-- 2. Insert Sample Data
INSERT INTO Orders (OrderID, OrderAmount, OrderDate)
VALUES 
    (1, '150', '2024-01-01'),      -- Valid number
    (2, '200USD', '2024-01-02'),   -- Invalid (contains letters)
    (3, '50', '2024-01-06'),    -- Valid decimal (but will fail if cast to INT)
    (4, 'Unknown', '2024-01-07'),  -- Invalid (pure text)
    (5, '10', '2024-01-10');       -- Valid number

SELECT * from Orders;

SELECT 
    OrderAmount AS OriginalValue,
    CAST(OrderAmount AS INT) AS CleanedAmount
FROM Orders;

SELECT 
    OrderAmount AS OriginalValue,
    TRY_CAST(OrderAmount AS INT) AS CleanedAmount
FROM Orders;

SELECT 
    OrderAmount AS OriginalValue,
    TRY_CONVERT(INT, OrderAmount) AS CleanedAmount
FROM Orders;

SELECT 
    OrderAmount AS OriginalValue,
    TRY_CAST(OrderAmount AS INT) AS CleanedAmount,
    DATENAME(weekday, OrderDate) AS OrderDay
FROM Orders;

SELECT 
    OrderAmount AS OriginalValue,
    TRY_CAST(OrderAmount AS INT) AS CleanedAmount,
    DATEPART(weekday, OrderDate) AS OrderDay
FROM Orders;
