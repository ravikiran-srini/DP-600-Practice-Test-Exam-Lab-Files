-- Create the Dimension Table for SCD Type 2
CREATE TABLE DimProduct (ID INT, Price INT, IsCurrent BIT);
CREATE TABLE Staging (ID INT, Price INT);

-- Initial Load (The 'Old' Data)
INSERT INTO DimProduct VALUES (101, 50, 1);

-- Staging Data (The 'New' Data)
INSERT INTO Staging VALUES (101, 70);

MERGE INTO DimProduct AS T
USING Staging AS S
ON T.ID = S.ID AND T.IsCurrent = 1
WHEN MATCHED AND T.Price <> S.Price
THEN UPDATE SET T.IsCurrent = 0;

INSERT INTO DimProduct (ID, Price, IsCurrent)
SELECT ID, Price, 1 FROM Staging;

SELECT * FROM DimProduct;

-- Create the Dimension Table for SCD Type 1
CREATE TABLE DimCustomer (ID INT, Email VARCHAR(50));

--Old value
INSERT INTO DimCustomer VALUES (500, 'old@email.com');

DECLARE @IncomingID INT = 500;
DECLARE @IncomingEmail VARCHAR(50) = 'new@email.com';

MERGE INTO DimCustomer AS Target
USING (SELECT @IncomingID AS ID, @IncomingEmail AS Email) AS Source ON Target.ID = Source.ID
WHEN MATCHED AND Target.Email <> Source.Email THEN
    UPDATE SET Target.Email = Source.Email
WHEN NOT MATCHED THEN
    INSERT (ID, Email) VALUES (Source.ID, Source.Email);

SELECT * FROM DimCustomer
