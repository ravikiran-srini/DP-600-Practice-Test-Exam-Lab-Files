-- 1. Create the Customers Table (For CLS Observation)
CREATE TABLE Customers (
    CustomerID INT,
    CustomerName VARCHAR(100),
    Email VARCHAR(100),
    NationalID VARCHAR(20),       -- Sensitive Column
    CreditCardNumber VARCHAR(20)   -- Sensitive Column
);

-- 2. Create the Subscriptions Table (For RLS Observation)
CREATE TABLE Subscriptions (
    SubscriptionID INT,
    CustomerID INT,
    PlanName VARCHAR(50),
    MonthlyRate DECIMAL(10,2),
    BranchRegion VARCHAR(50)       -- Filter Column for RLS
);

-- CLS: Explicitly lock down the sensitive columns for this user
DENY SELECT ON dbo.Customers(NationalID) TO [clientone@ravikiransrinivasulu.com];
DENY SELECT ON dbo.Customers(CreditCardNumber) TO [clientone@ravikiransrinivasulu.com];

--RLS:

-- 1. Create a separate schema to hold the security function
CREATE SCHEMA Security;
GO

-- 2. Create the inline table-valued function for filtering
CREATE FUNCTION 
Security.fn_subscriptionSecurityPredicate(@BranchRegion AS VARCHAR(50))
    RETURNS TABLE
WITH SCHEMABINDING
AS
    -- Added "AS result" to explicitly name the column
    RETURN SELECT 1 AS result 
    WHERE USER_NAME() <> 'clientone@ravikiransrinivasulu.com' OR @BranchRegion = 'North';
GO

-- 3. Apply the Security Policy to the Subscriptions table
CREATE SECURITY POLICY Security.SubscriptionFilter
ADD FILTER PREDICATE Security.fn_subscriptionSecurityPredicate(BranchRegion)
ON dbo.Subscriptions WITH (STATE = ON);
GO
